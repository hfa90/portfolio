Configurações do Negócio

📋 SQL de configuração — Supabase

Execute no SQL Editor do Supabase para criar/atualizar as tabelas necessárias:

-- Habilitar Realtime nas tabelas
alter publication supabase_realtime add table orders;

-- Tabela de fidelidade
create table if not exists loyalty_customers (
  id bigserial primary key,
  name text not null,
  phone text not null unique,
  points integer default 0,
  total_orders integer default 0,
  total_spent numeric default 0,
  total_redeemed integer default 0,
  created_at timestamptz default now(),
  updated_at timestamptz default now()
);

-- Histórico de pontos (opcional)
create table if not exists loyalty_history (
  id bigserial primary key,
  customer_id bigint references loyalty_customers(id),
  order_id bigint,
  points_earned integer,
  balance_after integer,
  description text,
  created_at timestamptz default now()
);

-- RLS: permitir leitura/escrita anônima (ajuste conforme sua política)
alter table loyalty_customers enable row level security;
create policy "allow_all" on loyalty_customers for all using (true) with check (true);
alter table loyalty_history enable row level security;
create policy "allow_all" on loyalty_history for all using (true) with check (true);
